import Security
import Foundation

// MARK: - KeychainError

public enum KeychainError: Error, LocalizedError {
    case saveFailed(OSStatus)
    case loadFailed(OSStatus)
    case deleteFailed(OSStatus)
    case dataDecodingFailed

    public var errorDescription: String? {
        switch self {
        case .saveFailed(let s):   return "Keychain save failed (OSStatus \(s))"
        case .loadFailed(let s):   return "Keychain load failed (OSStatus \(s))"
        case .deleteFailed(let s): return "Keychain delete failed (OSStatus \(s))"
        case .dataDecodingFailed:  return "Keychain data could not be decoded as UTF-8"
        }
    }
}

// MARK: - KeychainCredentialStore

/// Stores SMB credentials as internet passwords in the system Keychain.
///
/// Key: (server, account). Value: password (stored as Data, never in GRDB).
/// Uses Security.framework SecItem API directly — no third-party dependency.
public struct KeychainCredentialStore: Sendable {

    public init() {}

    /// Save or update a password for (server, account).
    /// Uses delete-then-add pattern so it is idempotent.
    public func save(server: String, account: String, password: String) throws {
        guard let passwordData = password.data(using: .utf8) else {
            throw KeychainError.dataDecodingFailed
        }
        let query: [String: Any] = [
            kSecClass as String:       kSecClassInternetPassword,
            kSecAttrServer as String:  server,
            kSecAttrAccount as String: account,
            kSecValueData as String:   passwordData
        ]
        // Delete existing first (update pattern — avoids errSecDuplicateItem)
        SecItemDelete(query as CFDictionary)
        let status = SecItemAdd(query as CFDictionary, nil)
        guard status == errSecSuccess else {
            throw KeychainError.saveFailed(status)
        }
    }

    /// Load the password for (server, account). Throws if not found.
    public func load(server: String, account: String) throws -> String {
        let query: [String: Any] = [
            kSecClass as String:            kSecClassInternetPassword,
            kSecAttrServer as String:       server,
            kSecAttrAccount as String:      account,
            kSecReturnData as String:       true,
            kSecMatchLimit as String:       kSecMatchLimitOne
        ]
        var result: AnyObject?
        let status = SecItemCopyMatching(query as CFDictionary, &result)
        guard status == errSecSuccess,
              let data = result as? Data,
              let password = String(data: data, encoding: .utf8) else {
            throw KeychainError.loadFailed(status)
        }
        return password
    }

    /// Delete credentials for (server, account). No-op if not found.
    public func delete(server: String, account: String) throws {
        let query: [String: Any] = [
            kSecClass as String:       kSecClassInternetPassword,
            kSecAttrServer as String:  server,
            kSecAttrAccount as String: account
        ]
        let status = SecItemDelete(query as CFDictionary)
        guard status == errSecSuccess || status == errSecItemNotFound else {
            throw KeychainError.deleteFailed(status)
        }
    }
}
```

import Foundation
import NetFS

// MARK: - SMBError

public enum SMBError: Error, LocalizedError {
    case invalidURL
    case mountFailed(status: Int)
    case timeout

    public var errorDescription: String? {
        switch self {
        case .invalidURL:          return "Could not construct SMB URL from host/share"
        case .mountFailed(let s):  return "SMB mount failed (NetFS status \(s))"
        case .timeout:             return "SMB connection timed out after 10 seconds"
        }
    }
}

// MARK: - NetFSMounter

/// Wraps NetFSMountURLAsync in async/await with a 10-second timeout.
///
/// Non-sandboxed only — pass nil for mountPath to let NetFS manage /Volumes automatically.
/// Checks for an existing mount with the same host/share before calling NetFS
/// (macOS silently reuses existing mounts, so this avoids a redundant callback).
public struct NetFSMounter: Sendable {

    public init() {}

    /// Mount an SMB share and return the mount point URL.
    ///
    /// - Parameters:
    ///   - host: Hostname or IP, e.g. "192.168.1.10" or "nas.local"
    ///   - share: Share name, e.g. "Backups"
    ///   - username: SMB username
    ///   - password: SMB password (loaded from Keychain by caller — never stored here)
    ///   - protocol: URL scheme, default "smb". Pass "nfs" for NFS shares.
    /// - Returns: The mount point URL in /Volumes
    /// - Throws: SMBError on invalid URL, mount failure, or 10-second timeout
    public func mountShare(
        host: String,
        share: String,
        username: String,
        password: String,
        protocol urlScheme: String = "smb"
    ) async throws -> URL {
        // Build share URL via URLComponents (handles percent-encoding of special chars)
        var components = URLComponents()
        components.scheme = urlScheme
        components.host = host
        components.path = "/\(share)"
        guard let shareURL = components.url else {
            throw SMBError.invalidURL
        }

        // Check if already mounted — macOS reuses existing mounts silently
        if let existing = findExistingMount(host: host, share: share) {
            return existing
        }

        // Mount with 10-second timeout using task cancellation
        return try await withTimeout(seconds: 10, operation: {
            try await mountAsync(shareURL: shareURL, username: username, password: password)
        })
    }

    /// Unmount a share at the given URL. Logs and ignores errors (unmount can fail silently on macOS).
    public func unmount(at url: URL) {
        // NSWorkspace.unmountAndEjectDevice is AppKit — use FileManager unmount
        // Non-fatal: log the failure but never throw
        try? FileManager.default.unmountVolume(at: url, options: .allPartitionsAndEjectDisk, completionHandler: { _ in })
    }

    // MARK: - Private

    private func findExistingMount(host: String, share: String) -> URL? {
        let volumes = FileManager.default.mountedVolumeURLs(
            includingResourceValuesForKeys: [.volumeNameKey],
            options: [.skipHiddenVolumes]
        ) ?? []

        // NetFS typically mounts as /Volumes/{share} or /Volumes/{share} N
        // Match by checking volume name against share name
        let shareLower = share.lowercased()
        for url in volumes {
            let name = (try? url.resourceValues(forKeys: [.volumeNameKey]))?.volumeName ?? ""
            if name.lowercased() == shareLower || name.lowercased().hasPrefix(shareLower) {
                return url
            }
        }
        return nil
    }

    private func mountAsync(shareURL: URL, username: String, password: String) async throws -> URL {
        return try await withCheckedThrowingContinuation { continuation in
            var requestID: AsyncRequestID?
            let result = NetFSMountURLAsync(
                shareURL as CFURL,
                nil,                     // mountPath: nil — NetFS manages /Volumes (non-sandboxed)
                username as CFString,
                password as CFString,
                nil,                     // openOptions
                nil,                     // mountOptions
                &requestID,
                .main,
                { status, mountpoints in
                    if status == 0,
                       let mounts = mountpoints as? [String],
                       let first = mounts.first {
                        continuation.resume(returning: URL(fileURLWithPath: first))
                    } else {
                        continuation.resume(throwing: SMBError.mountFailed(status: Int(status)))
                    }
                }
            )
            if result != 0 {
                continuation.resume(throwing: SMBError.mountFailed(status: Int(result)))
            }
        }
    }
}

// MARK: - Timeout helper

/// Run an async operation with a deadline. Throws SMBError.timeout if it does not complete in time.
private func withTimeout<T: Sendable>(seconds: Double, operation: @escaping @Sendable () async throws -> T) async throws -> T {
    try await withThrowingTaskGroup(of: T.self) { group in
        group.addTask { try await operation() }
        group.addTask {
            try await Task.sleep(for: .seconds(seconds))
            throw SMBError.timeout
        }
        let result = try await group.next()!
        group.cancelAll()
        return result
    }
}
```

Note on Swift 6 concurrency: `NetFSMountURLAsync` takes a C block callback. The closure bridging may produce a Sendable warning. If the compiler rejects the closure in `mountAsync`, annotate the closure parameter with `@Sendable` or wrap the body in `nonisolated(unsafe)`. The `withCheckedThrowingContinuation` pattern is the documented approach for bridging C callbacks in Swift concurrency.
