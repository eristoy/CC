**BackupJob construction for Phase 2:**

For Phase 2, use the watch folder as both source and a temporary local destination at `~/Library/Application Support/AbletonBackup/Backup/`. Insert a Project and DestinationConfig into GRDB on first launch if none exist (bootstrap). This lets the engine run end-to-end in Phase 2 without a settings UI.

**Concurrent deduplication:** BackupEngine actor already handles per-project deduplication (Task join). BackupCoordinator's `guard case .idle = status else { return }` prevents overlapping status transitions at the coordinator level.

**Swift 6 FSEvents bridge:** The FSEventsWatcher callback is `(String) -> Void` and is called on the main thread. Inside BackupCoordinator.startWatching(), capture as:
```swift
watcher = FSEventsWatcher(url: folder) { [weak self] path in
    Task { @MainActor [weak self] in
        await self?.handleALSChange(at: path)
    }
}
```
This is valid Swift 6: the outer closure is nonisolated (no @MainActor captures), the inner Task explicitly hops to @MainActor. No Sendable violation.

```swift
// AbletonBackup/BackupCoordinator.swift

import Foundation
import BackupEngine
import GRDB

// MARK: - BackupStatus

public enum BackupStatus: Sendable {
    case idle
    case running
    case error(String)

    public var iconName: String {
        switch self {
        case .idle:    return "waveform"
        case .running: return "arrow.triangle.2.circlepath"
        case .error:   return "exclamationmark.triangle.fill"
        }
    }
}

// MARK: - BackupTrigger

public enum BackupTrigger: Sendable {
    case fsEvent
    case scheduled
    case manual
}

// MARK: - BackupCoordinator

@Observable
@MainActor
public final class BackupCoordinator {

    // MARK: - Observable state (drives menu bar icon + menu content)

    public var status: BackupStatus = .idle
    public var lastBackupAt: Date? = nil
    public var statusIcon: String { status.iconName }

    // MARK: - Login item (APP-03)

    public let loginItemManager = LoginItemManager()

    // MARK: - Private

    private var engine: BackupEngine?
    private var db: AppDatabase?
    private var watchedProjectsFolder: URL?
    private var watcher: FSEventsWatcher?
    private let scheduler = SchedulerTask()

    // Phase 2 bootstrap: single project + local destination in App Support
    private var bootstrapProject: Project?
    private var bootstrapDestID: String?

    // MARK: - Init

    public init() {
        Task { @MainActor [weak self] in
            await self?.setup()
        }
    }

    // MARK: - Setup

    private func setup() async {
        // 1. Request notification permission
        NotificationService.requestAuthorization()

        // 2. Open/create GRDB database in Application Support
        do {
            let appSupport = try FileManager.default.url(
                for: .applicationSupportDirectory,
                in: .userDomainMask,
                appropriateFor: nil,
                create: true
            ).appendingPathComponent("AbletonBackup")
            try FileManager.default.createDirectory(at: appSupport, withIntermediateDirectories: true)
            let dbPath = appSupport.appendingPathComponent("abletonbackup.db").path
            let database = try AppDatabase.makeShared(at: dbPath)
            self.db = database

            // 3. Initialize BackupEngine (no adapters needed for Phase 2 bootstrap)
            //    Phase 3 will wire real DestinationConfigs from settings UI.
            self.engine = BackupEngine(db: database, adapters: [])

        } catch {
            status = .error("Database setup failed: \(error.localizedDescription)")
            return
        }

        // 4. Discover Ableton Projects folder (DISC-01)
        let folder = AbletonPrefsReader.discoverProjectsFolder()
        self.watchedProjectsFolder = folder

        // 5. Start FSEvents watcher (TRIG-01)
        if let folder {
            startWatching(folder: folder)
        }

        // 6. Start scheduled backup loop (TRIG-02)
        scheduler.start(interval: SchedulerTask.defaultInterval) { [weak self] in
            await self?.runBackup(trigger: .scheduled)
        }
    }

    // MARK: - FSEvents (TRIG-01)

    private func startWatching(folder: URL) {
        // Outer closure is nonisolated (required by FSEventsWatcher).
        // Inner Task hops to @MainActor for coordinator state access.
        // Swift 6: valid because the Task closure explicitly captures @MainActor isolation.
        watcher = FSEventsWatcher(url: folder) { [weak self] path in
            Task { @MainActor [weak self] in
                await self?.handleALSChange(at: path)
            }
        }
    }

    private func handleALSChange(at path: String) async {
        // FSEventsWatcher already filters to .als — double-check for safety
        guard path.hasSuffix(".als") else { return }
        await runBackup(trigger: .fsEvent)
    }

    // MARK: - Backup execution (TRIG-01, TRIG-02, TRIG-03)

    /// Run a backup. Guards against concurrent runs via status check.
    /// BackupEngine.runJob() handles per-project deduplication internally.
    public func runBackup(trigger: BackupTrigger) async {
        guard case .idle = status else { return }
        guard let engine else { return }

        status = .running
        let projectName = watchedProjectsFolder?.lastPathComponent ?? "Ableton Projects"

        do {
            // Phase 2 bootstrap: if no Project/Destination exists in DB yet,
            // create a minimal local-to-local job using the watch folder.
            // Phase 3 replaces this with real settings-configured jobs.
            let job = try await buildBootstrapJob()
            _ = try await engine.runJob(job)
            status = .idle
            lastBackupAt = Date()
            NotificationService.sendBackupSuccess(projectName: projectName)  // NOTIF-01
        } catch {
            let msg = error.localizedDescription
            status = .error(msg)
            NotificationService.sendBackupFailure(projectName: projectName, error: msg)  // NOTIF-02
        }
    }

    /// Build a minimal BackupJob for Phase 2.
    /// Uses watched projects folder as source + App Support backup folder as destination.
    /// Phase 3 replaces this with real multi-destination, multi-project job construction.
    private func buildBootstrapJob() async throws -> BackupJob {
        guard let db, let watchedProjectsFolder else {
            throw CoordinatorError.notConfigured
        }

        let appSupport = try FileManager.default.url(
            for: .applicationSupportDirectory,
            in: .userDomainMask,
            appropriateFor: nil,
            create: true
        ).appendingPathComponent("AbletonBackup")
        let backupDir = appSupport.appendingPathComponent("Backup")
        try FileManager.default.createDirectory(at: backupDir, withIntermediateDirectories: true)

        // Upsert bootstrap Project in GRDB if not yet present
        let projectID = "bootstrap-project"
        let project = Project(
            id: projectID,
            name: watchedProjectsFolder.lastPathComponent,
            rootURL: watchedProjectsFolder
        )
        try await db.pool.write { db in
            try project.save(db)
        }

        // Upsert bootstrap DestinationConfig (local, App Support backup folder)
        let destID = "bootstrap-local"
        let dest = DestinationConfig(
            id: destID,
            type: "local",
            label: "App Support Backup",
            rootURL: backupDir,
            retentionCount: 10
        )
        try await db.pool.write { db in
            try dest.save(db)
        }

        // Add LocalDestinationAdapter if engine doesn't have it yet
        // (idempotent — Phase 3 wires real adapters via settings)
        let adapter = LocalDestinationAdapter(config: dest)
        await engine?.addAdapter(adapter)

        return BackupJob(project: project, destinationIDs: [destID])
    }
}

// MARK: - BackupEngine extension for dynamic adapter addition

// Add addAdapter() to BackupEngine actor for Phase 2 bootstrap.
// Phase 3 replaces with full settings-driven adapter initialization.
extension BackupEngine {
    func addAdapter(_ adapter: some DestinationAdapter) {
        adapters[adapter.id] = adapter
    }
}

// MARK: - Errors

enum CoordinatorError: Error, LocalizedError {
    case notConfigured

    var errorDescription: String? {
        switch self {
        case .notConfigured: return "Watch folder or database not configured"
        }
    }
}
```

// AbletonBackup/Views/MenuBarView.swift

import SwiftUI
import ServiceManagement

/// The dropdown content for the AbletonBackup menu bar item.
struct MenuBarView: View {
    @Environment(BackupCoordinator.self) private var coordinator
    @State private var loginItemError: String? = nil

    var body: some View {
        VStack(alignment: .leading, spacing: 0) {
            // Status header
            statusSection
            Divider()

            // Backup action (TRIG-03)
            backupSection
            Divider()

            // Settings
            settingsSection
            Divider()

            // Quit
            Button("Quit AbletonBackup") {
                NSApplication.shared.terminate(nil)
            }
            .keyboardShortcut("q")
        }
        .padding(.vertical, 4)
    }

    // MARK: - Status

    private var statusSection: some View {
        VStack(alignment: .leading, spacing: 2) {
            HStack {
                Image(systemName: coordinator.statusIcon)
                    .foregroundStyle(statusColor)
                Text(statusText)
                    .fontWeight(.medium)
            }
            .padding(.horizontal, 12)
            .padding(.vertical, 6)

            if let lastAt = coordinator.lastBackupAt {
                Text("Last backup: \(lastAt.formatted(.relative(presentation: .named)))")
                    .font(.caption)
                    .foregroundStyle(.secondary)
                    .padding(.horizontal, 12)
                    .padding(.bottom, 4)
            } else {
                Text("Never backed up")
                    .font(.caption)
                    .foregroundStyle(.secondary)
                    .padding(.horizontal, 12)
                    .padding(.bottom, 4)
            }
        }
    }

    // MARK: - Backup actions

    private var backupSection: some View {
        VStack(alignment: .leading, spacing: 0) {
            Button("Back Up Now") {
                Task { await coordinator.runBackup(trigger: .manual) }
            }
            .disabled(coordinator.status == .running)
            .padding(.horizontal, 12)
            .padding(.vertical, 6)
        }
    }

    // MARK: - Settings

    private var settingsSection: some View {
        VStack(alignment: .leading, spacing: 0) {
            // Launch at Login toggle (APP-03)
            Toggle(isOn: loginItemBinding) {
                Text("Launch at Login")
            }
            .padding(.horizontal, 12)
            .padding(.vertical, 6)

            // Show approval hint if SMAppService requires it
            if coordinator.loginItemManager.status == .requiresApproval {
                VStack(alignment: .leading, spacing: 4) {
                    Text("Approval required in System Settings")
                        .font(.caption)
                        .foregroundStyle(.orange)
                        .padding(.horizontal, 12)
                    Button("Open System Settings…") {
                        coordinator.loginItemManager.openSystemSettings()
                    }
                    .font(.caption)
                    .padding(.horizontal, 12)
                    .padding(.bottom, 4)
                }
            }

            if let error = loginItemError {
                Text(error)
                    .font(.caption)
                    .foregroundStyle(.red)
                    .padding(.horizontal, 12)
                    .padding(.bottom, 4)
            }
        }
    }

    // MARK: - Helpers

    private var loginItemBinding: Binding<Bool> {
        Binding(
            get: { coordinator.loginItemManager.isEnabled },
            set: { newValue in
                loginItemError = nil
                do {
                    try coordinator.loginItemManager.setEnabled(newValue)
                } catch {
                    loginItemError = error.localizedDescription
                }
            }
        )
    }

    private var statusColor: Color {
        switch coordinator.status {
        case .idle:    return .green
        case .running: return .blue
        case .error:   return .red
        }
    }

    private var statusText: String {
        switch coordinator.status {
        case .idle:         return "Idle"
        case .running:      return "Backing up…"
        case .error(let e): return "Error: \(e)"
        }
    }
}

// MARK: - BackupStatus Equatable for disabled state check

extension BackupStatus: Equatable {
    public static func == (lhs: BackupStatus, rhs: BackupStatus) -> Bool {
        switch (lhs, rhs) {
        case (.idle, .idle), (.running, .running): return true
        case (.error(let a), .error(let b)):        return a == b
        default:                                    return false
        }
    }
}
