Create `AbletonBackup/DestinationManager.swift`. This is an @Observable @MainActor class (not actor — @Observable requires class) that:

1. Holds per-destination status keyed by destination ID
2. Runs a 30-second polling loop calling probe() on every adapter
3. Exposes an adapter factory to build the right adapter type from DestinationConfig

```swift
import Foundation
import BackupEngine
import Network
import AppKit
import OSLog

// MARK: - DestinationManager

/// Manages the lifecycle of all configured backup destinations.
///
/// Responsibilities:
/// - Build correct DestinationAdapter from DestinationConfig.type
/// - Poll all destinations every 30 seconds via probe() and update statuses
/// - Reconnect SMB destinations after system wakes from sleep
/// - Expose per-destination status as @Observable state for SwiftUI
///
/// Placed in the app target (not BackupEngine) to access AppKit NSWorkspace.
@Observable
@MainActor
final class DestinationManager {

    // MARK: - Observable state

    /// Per-destination availability status. Key = DestinationConfig.id.
    var statuses: [String: DestinationStatus] = [:]

    // MARK: - Private

    private var adapters: [String: any DestinationAdapter] = [:]  // keyed by destination ID
    private var pollingTask: Task<Void, Never>?
    private let logger = Logger(subsystem: "com.abletonbackup", category: "DestinationManager")

    // MARK: - Init

    init() {}

    // MARK: - Setup

    /// Load all destinations from GRDB and start polling. Called during BackupCoordinator.setup().
    func setup(configs: [DestinationConfig]) {
        // Build adapter for each config
        for config in configs {
            adapters[config.id] = makeAdapter(for: config)
            statuses[config.id] = .unavailable(reason: "Not yet probed")
        }
        logger.info("DestinationManager: loaded \(configs.count) destination(s)")
        startPolling()
        registerWakeObserver()
    }

    // MARK: - Adapter management

    /// Add a newly created destination (after user saves from settings UI).
    func addDestination(config: DestinationConfig) {
        adapters[config.id] = makeAdapter(for: config)
        statuses[config.id] = .unavailable(reason: "Not yet probed")
        Task {
            await probeOne(id: config.id)
        }
    }

    /// Remove a destination (after user deletes from settings UI).
    func removeDestination(id: String) {
        adapters.removeValue(forKey: id)
        statuses.removeValue(forKey: id)
    }

    /// Return all currently loaded adapters (used by BackupCoordinator.runBackup).
    func availableAdapters() -> [any DestinationAdapter] {
        adapters.values.filter { adapter in
            if case .available = statuses[adapter.id] ?? .unavailable(reason: "") { return true }
            return false
        }
    }

    /// Return adapter for a specific destination ID.
    func adapter(for id: String) -> (any DestinationAdapter)? {
        adapters[id]
    }

    // MARK: - Polling (DEST-06)

    func startPolling() {
        pollingTask?.cancel()
        pollingTask = Task { [weak self] in
            while !Task.isCancelled {
                await self?.probeAll()
                try? await Task.sleep(for: .seconds(30))
            }
        }
    }

    func stopPolling() {
        pollingTask?.cancel()
        pollingTask = nil
    }

    private func probeAll() async {
        for (id, adapter) in adapters {
            let status = await adapter.probe()
            statuses[id] = status
            switch status {
            case .available:
                logger.debug("probe [\(id)]: available")
            case .unavailable(let reason):
                logger.warning("probe [\(id)]: unavailable — \(reason, privacy: .public)")
            }
        }
    }

    private func probeOne(id: String) async {
        guard let adapter = adapters[id] else { return }
        statuses[id] = await adapter.probe()
    }

    // MARK: - Sleep/wake reconnection (DEST-07)

    private func registerWakeObserver() {
        NSWorkspace.shared.notificationCenter.addObserver(
            forName: NSWorkspace.didWakeNotification,
            object: nil,
            queue: .main
        ) { [weak self] _ in
            Task { @MainActor [weak self] in
                self?.logger.info("DestinationManager: system woke — scheduling reconnect")
                await self?.reconnectNetworkDestinations()
            }
        }
    }

    /// Reconnect SMB destinations after wake. Waits for network to become available first.
    func reconnectNetworkDestinations() async {
        // Wait for network path to become satisfied (up to 15 seconds)
        await waitForNetwork(timeout: 15)
        // Brief additional delay for SMB server availability
        try? await Task.sleep(for: .seconds(2))

        for (id, adapter) in adapters {
            guard let smbAdapter = adapter as? SMBDestinationAdapter else { continue }
            do {
                let mountURL = try await smbAdapter.ensureMounted()
                logger.info("DestinationManager: reconnected \(id) at \(mountURL.path, privacy: .public)")
                statuses[id] = await smbAdapter.probe()
            } catch {
                logger.warning("DestinationManager: reconnect failed for \(id) — \(error.localizedDescription, privacy: .public)")
                statuses[id] = .unavailable(reason: error.localizedDescription)
            }
        }
        // Re-probe all destinations (including non-SMB) after wake
        await probeAll()
    }

    // MARK: - Adapter factory

    private func makeAdapter(for config: DestinationConfig) -> any DestinationAdapter {
        switch config.type {
        case .local:
            return LocalDestinationAdapter(config: config)
        case .mountedVolume:
            return MountedVolumeAdapter(config: config)
        case .nas:
            return SMBDestinationAdapter(config: config)
        case .icloud:
            return iCloudDestinationAdapter(config: config)
        case .github:
            // Phase 6 — return LocalDestinationAdapter as stub; GitHub destination ships in Phase 6
            return LocalDestinationAdapter(config: config)
        }
    }
}

// MARK: - Network readiness helper

/// Wait until NWPathMonitor reports network is satisfied, or until timeout.
@MainActor
private func waitForNetwork(timeout: Double) async {
    let monitor = NWPathMonitor()
    let stream = AsyncStream<NWPath.Status> { continuation in
        monitor.pathUpdateHandler = { path in
            continuation.yield(path.status)
        }
        monitor.start(queue: .global(qos: .utility))
        continuation.onTermination = { _ in monitor.cancel() }
    }
    let deadline = Date.now.addingTimeInterval(timeout)
    for await status in stream {
        if status == .satisfied { break }
        if Date.now >= deadline { break }
    }
    monitor.cancel()
}
```
