<!-- Verification step: checksum comparison after copy -->
After transfer() returns [BackupFileRecord] (each with a checksum computed during copy),
BackupEngine verifies by re-reading each source file's checksum and comparing.
Wait — research says: "verify manifest against destination after backup."
Correct approach: verify the DESTINATION file matches the SOURCE checksum stored in BackupFileRecord.
  - BackupFileRecord.checksum was computed from DESTINATION during copy (inline during copy)
  - Verification step: re-compute checksum from DESTINATION and compare to BackupFileRecord.checksum
  - If mismatch: markCorrupt (disk write error corrupted the destination copy)
  - If match: markVerified
This detects silent write corruption. NOT about comparing source vs destination (the file was
potentially modified on disk between copy and verify, which is fine — we protect the destination).

<!-- Incremental logic in BackupEngine -->
BackupEngine must fetch the previous version's BackupFileRecord list before resolving files.
For each FileEntry from ProjectResolver.resolve(), call:
  ProjectResolver.needsCopy(entry: fileEntry, previousRecord: previousManifestRecord)
Skip files where needsCopy returns false (pass nil record for new files).
Only pass the needsCopy=true files to adapter.transfer().
Record skipped files in BackupJobResult.filesSkipped.

<!-- Job deduplication (concurrent runJob calls) -->
The BackupEngine actor holds a dictionary: [String: Task<BackupJobResult, Error>]
(key = projectID). If a task already exists for that project, return its value (join it)
instead of starting a new job. This prevents concurrent duplicate jobs for the same project.

```swift
import Foundation
import GRDB

public actor BackupEngine {
    private let db: AppDatabase
    private var adapters: [String: any DestinationAdapter]  // keyed by destinationID
    private var runningJobs: [String: Task<BackupJobResult, Error>]  // keyed by projectID

    public init(db: AppDatabase, adapters: [any DestinationAdapter]) {
        self.db = db
        self.adapters = Dictionary(uniqueKeysWithValues: adapters.map { ($0.id, $0) })
        self.runningJobs = [:]
    }

    public func runJob(_ job: BackupJob) async throws -> BackupJobResult {
        // Deduplication: if already running for this project, join it
        if let existing = runningJobs[job.project.id] {
            return try await existing.value
        }

        let task = Task {
            try await self.executeJob(job)
        }
        runningJobs[job.project.id] = task
        defer { runningJobs.removeValue(forKey: job.project.id) }
        return try await task.value
    }

    private func executeJob(_ job: BackupJob) async throws -> BackupJobResult {
        let project = job.project
        let targetAdapters = job.destinationIDs.compactMap { adapters[$0] }
        guard !targetAdapters.isEmpty else {
            throw BackupEngineError.noDestinationsConfigured
        }

        // Resolve files
        let allFiles = try ProjectResolver.resolve(at: URL(fileURLWithPath: project.path))

        // Fetch previous manifest for incremental comparison
        let previousRecords = try await fetchPreviousManifest(projectID: project.id)
        let previousByPath = Dictionary(uniqueKeysWithValues: previousRecords.map { ($0.relativePath, $0) })

        // Incremental filter
        var filesToCopy: [FileEntry] = []
        var skippedCount = 0
        for file in allFiles {
            if ProjectResolver.needsCopy(entry: file, previousRecord: previousByPath[file.relativePath]) {
                filesToCopy.append(file)
            } else {
                skippedCount += 1
            }
        }

        // Create pending version records (one per destination)
        let versionID = VersionManager.newVersionID()
        for adapter in targetAdapters {
            let version = BackupVersion(
                id: versionID,
                projectID: project.id,
                destinationID: adapter.id,
                status: .pending,
                fileCount: nil,
                totalBytes: nil,
                createdAt: Date(),
                completedAt: nil,
                errorMessage: nil
            )
            try await db.pool.write { db in try version.insert(db) }
        }

        // Transition to copying
        try await updateVersionStatus(versionID: versionID, status: .copying)

        // Fan out to destinations concurrently via TaskGroup
        var destinationResults: [DestinationResult] = []
        try await withThrowingTaskGroup(of: DestinationResult.self) { group in
            for adapter in targetAdapters {
                group.addTask {
                    do {
                        let records = try await adapter.transfer(filesToCopy, versionID: versionID) { _ in }
                        // Insert BackupFileRecord rows
                        try await self.db.pool.write { db in
                            for record in records { try record.insert(db) }
                        }
                        return DestinationResult(
                            destinationID: adapter.id,
                            status: .copy_complete,
                            errorMessage: nil
                        )
                    } catch {
                        return DestinationResult(
                            destinationID: adapter.id,
                            status: .corrupt,
                            errorMessage: error.localizedDescription
                        )
                    }
                }
            }
            for try await result in group {
                destinationResults.append(result)
            }
        }

        // Build manifest from the transfer results
        let allRecords = try await db.pool.read { db in
            try BackupFileRecord
                .filter(Column("versionID") == versionID)
                .fetchAll(db)
        }
        let manifest = BackupManifest(
            versionID: versionID,
            projectID: project.id,
            destinationID: targetAdapters.first!.id,
            files: allRecords.compactMap { rec in
                rec.checksum.map { ManifestEntry(relativePath: rec.relativePath,
                                                  size: rec.sourceSize, checksum: $0) }
            }
        )

        // Finalize copy_complete status
        for adapter in targetAdapters {
            try await VersionManager.finalizeCopy(versionID: versionID, manifest: manifest, db: db.pool)
        }

        // Verification pass: re-read destination checksums and compare
        try await updateVersionStatus(versionID: versionID, status: .verifying)
        var allVerified = true
        var corruptReason: String? = nil
        // Re-read each BackupFileRecord.checksum (set during copy) from DB
        // Re-compute from destination and compare
        // NOTE: For Phase 1 (local), the checksum WAS computed from destination during copy.
        // Verification = trust the inline checksum (no second read needed).
        // HOWEVER: implement a spot-check verify to detect the scenario where the destination
        // file was corrupted after copy (e.g., disk write error surfaced after close).
        // Re-read one file per version to confirm the checksum still matches.
        // Full re-verification of every file is too slow for Phase 1; spot-check is the pragmatic choice.
        // TODO: Phase 3 can add a "deep verify" option to the settings UI.

        // For correctness in the integration test: full verification (re-read all destination checksums)
        let versionDir = URL(fileURLWithPath: targetAdapters[0].config.rootPath)
            .appendingPathComponent(versionID)
        for record in allRecords {
            guard let storedChecksum = record.checksum else { continue }
            let destFile = versionDir.appendingPathComponent(record.relativePath)
            let actualChecksum = try FileCopyPipeline.computeChecksum(of: destFile)
            if actualChecksum != storedChecksum {
                allVerified = false
                corruptReason = "Checksum mismatch for \(record.relativePath)"
                break
            }
        }

        if allVerified {
            try await VersionManager.markVerified(versionID: versionID, db: db.pool)
        } else {
            try await VersionManager.markCorrupt(
                versionID: versionID,
                reason: corruptReason ?? "Unknown checksum mismatch",
                db: db.pool
            )
        }

        // Pruning pass — only after new version is verified
        if allVerified {
            for adapter in targetAdapters {
                _ = try await VersionManager.pruneOldVersions(
                    for: project.id,
                    destinationID: adapter.id,
                    retentionCount: adapter.config.retentionCount,
                    destinationRootPath: adapter.config.rootPath,
                    db: db.pool
                )
            }
        }

        let finalStatus: VersionStatus = allVerified ? .verified : .corrupt
        return BackupJobResult(
            versionID: versionID,
            projectID: project.id,
            filesCopied: filesToCopy.count,
            filesSkipped: skippedCount,
            totalBytes: manifest.totalBytes,
            status: finalStatus,
            destinationResults: destinationResults
        )
    }

    private func updateVersionStatus(_ versionID: String, status: VersionStatus) async throws {
        try await db.pool.write { db in
            try db.execute(
                sql: "UPDATE backupVersion SET status = ? WHERE id = ?",
                arguments: [status.rawValue, versionID]
            )
        }
    }

    private func fetchPreviousManifest(projectID: String) async throws -> [BackupFileRecord] {
        try await db.pool.read { db in
            // Find the most recent verified version for this project
            guard let latestVersion = try BackupVersion
                .filter(Column("projectID") == projectID)
                .filter(Column("status") == VersionStatus.verified.rawValue)
                .order(Column("createdAt").desc)
                .fetchOne(db) else { return [] }

            return try BackupFileRecord
                .filter(Column("versionID") == latestVersion.id)
                .fetchAll(db)
        }
    }
}

    case noDestinationsConfigured
    case projectNotFound
}
```
