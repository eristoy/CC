**MountedVolumeAdapter.swift** (DEST-02):
```swift
import Foundation

/// Destination adapter for a NAS or external drive already mounted in /Volumes.
/// No credential entry — the volume is already authenticated by the OS.
/// Uses rootPath from DestinationConfig (the volume mount point, e.g. "/Volumes/MyNAS/Backups").
public struct MountedVolumeAdapter: DestinationAdapter {
    public let id: String
    public let config: DestinationConfig

    public init(config: DestinationConfig) {
        self.id = config.id
        self.config = config
    }

    public func transfer(
        _ files: [FileEntry],
        versionID: String,
        progress: @escaping @Sendable (TransferProgress) -> Void
    ) async throws -> [BackupFileRecord] {
        // Identical pattern to LocalDestinationAdapter
        let rootURL = URL(fileURLWithPath: config.rootPath)
        let versionDir = rootURL.appendingPathComponent(versionID)
        try FileManager.default.createDirectory(at: versionDir, withIntermediateDirectories: true)
        var records: [BackupFileRecord] = []
        records.reserveCapacity(files.count)
        for (index, file) in files.enumerated() {
            let destURL = versionDir.appendingPathComponent(file.relativePath)
            let checksum = try FileCopyPipeline.copyFileWithChecksum(source: file.url, destination: destURL)
            records.append(BackupFileRecord(
                rowid: nil, versionID: versionID, relativePath: file.relativePath,
                sourceMtime: file.mtime, sourceSize: Int(file.size), checksum: checksum, copied: true
            ))
            progress(TransferProgress(
                relativePath: file.relativePath, bytesCopied: file.size,
                totalFiles: files.count, completedFiles: index + 1
            ))
        }
        return records
    }

    public func pruneVersions(beyond retentionCount: Int, for projectID: String) async throws {
        // Orchestrated by VersionManager — no-op stub consistent with LocalDestinationAdapter
    }

    public func probe() async -> DestinationStatus {
        probeWriteAccess(at: config.rootPath)
    }
}
```

```swift
import Foundation
import OSLog

/// Destination adapter for a NAS accessible via direct SMB/NFS with stored credentials.
///
/// Lifecycle per backup:
/// 1. probe(): verify mount point is present and writable (no remount if already mounted)
/// 2. ensureMounted(): mount SMB share via NetFSMounter if not already in /Volumes
/// 3. transfer(): copy files via FileCopyPipeline (identical to LocalDestinationAdapter)
///
/// Credentials are loaded from Keychain at mount time using config.username as the account key.
/// Passwords are NEVER stored in DestinationConfig or GRDB.
public struct SMBDestinationAdapter: DestinationAdapter {

    public let id: String
    public let config: DestinationConfig

    private let credentialStore = KeychainCredentialStore()
    private let mounter = NetFSMounter()
    private let logger = Logger(subsystem: "com.abletonbackup", category: "SMBAdapter")

    public init(config: DestinationConfig) {
        self.id = config.id
        self.config = config
    }

    // MARK: - probe()

    /// Check if the SMB mount point (rootPath) is reachable and writable.
    /// Does NOT remount — DestinationManager (plan 04-04) handles remounting on wake.
    public func probe() async -> DestinationStatus {
        guard config.host != nil, config.share != nil, config.username != nil else {
            return .unavailable(reason: "Incomplete SMB configuration: host/share/username required")
        }
        return probeWriteAccess(at: config.rootPath)
    }

    // MARK: - ensureMounted()

    /// Mount the SMB share if not already present. Returns the mount point URL.
    /// Called by DestinationManager before backup and on wake reconnect.
    public func ensureMounted() async throws -> URL {
        guard let host = config.host, let share = config.share, let username = config.username else {
            throw SMBAdapterError.incompleteConfig
        }

        // Load credentials from Keychain (never from DestinationConfig)
        let password = try credentialStore.load(server: host, account: username)

        // Mount (NetFSMounter checks for existing mount first)
        let mountURL = try await mounter.mountShare(host: host, share: share, username: username, password: password)
        logger.info("SMBAdapter ensureMounted: \(host)/\(share) at \(mountURL.path, privacy: .public)")
        return mountURL
    }

    // MARK: - transfer()

    public func transfer(
        _ files: [FileEntry],
        versionID: String,
        progress: @escaping @Sendable (TransferProgress) -> Void
    ) async throws -> [BackupFileRecord] {
        let rootURL = URL(fileURLWithPath: config.rootPath)
        let versionDir = rootURL.appendingPathComponent(versionID)
        try FileManager.default.createDirectory(at: versionDir, withIntermediateDirectories: true)
        var records: [BackupFileRecord] = []
        records.reserveCapacity(files.count)
        for (index, file) in files.enumerated() {
            let destURL = versionDir.appendingPathComponent(file.relativePath)
            let checksum = try FileCopyPipeline.copyFileWithChecksum(source: file.url, destination: destURL)
            records.append(BackupFileRecord(
                rowid: nil, versionID: versionID, relativePath: file.relativePath,
                sourceMtime: file.mtime, sourceSize: Int(file.size), checksum: checksum, copied: true
            ))
            progress(TransferProgress(
                relativePath: file.relativePath, bytesCopied: file.size,
                totalFiles: files.count, completedFiles: index + 1
            ))
        }
        return records
    }

    // MARK: - pruneVersions()

    public func pruneVersions(beyond retentionCount: Int, for projectID: String) async throws {
        // Orchestrated by VersionManager — no-op stub
    }
}

// MARK: - SMBAdapterError

public enum SMBAdapterError: Error, LocalizedError {
    case incompleteConfig

    public var errorDescription: String? {
        "SMB destination is missing host, share, or username configuration"
    }
}
```
