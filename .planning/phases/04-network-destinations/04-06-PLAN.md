```swift
import SwiftUI
import BackupEngine

/// Sheet presented when user taps "+" in DestinationsSettingsView.
/// Shows a grouped type picker (Cloud / Network sections) then navigates to setup sub-view.
struct AddDestinationView: View {
    @Environment(BackupCoordinator.self) private var coordinator
    @Environment(\.dismiss) private var dismiss

    @State private var selectedType: DestinationType? = nil

    var body: some View {
        NavigationStack {
            List {
                Section("Cloud") {
                    Button(action: { selectedType = .icloud }) {
                        typeRow(icon: "icloud.fill", label: "iCloud Drive", subtitle: "Back up to iCloud Drive")
                    }
                    .buttonStyle(.plain)
                }
                Section("Network") {
                    Button(action: { selectedType = .mountedVolume }) {
                        typeRow(icon: "network", label: "Mounted Volume", subtitle: "NAS or external drive already mounted in /Volumes")
                    }
                    .buttonStyle(.plain)

                    Button(action: { selectedType = .nas }) {
                        typeRow(icon: "server.rack", label: "SMB / NFS Network Drive", subtitle: "Connect directly with credentials")
                    }
                    .buttonStyle(.plain)
                }
            }
            .navigationTitle("Add Destination")
            .toolbar {
                ToolbarItem(placement: .cancellationAction) {
                    Button("Cancel") { dismiss() }
                }
            }
            .navigationDestination(item: $selectedType) { type in
                destinationSetupView(for: type)
            }
        }
        .frame(minWidth: 380, minHeight: 320)
    }

    @ViewBuilder
    private func destinationSetupView(for type: DestinationType) -> some View {
        switch type {
        case .mountedVolume:
            MountedVolumeSetupView()
        case .nas:
            SMBSetupView()
        case .icloud:
            iCloudSetupView()
        default:
            Text("Not available")
        }
    }

    private func typeRow(icon: String, label: String, subtitle: String) -> some View {
        HStack(spacing: 12) {
            Image(systemName: icon)
                .font(.title2)
                .foregroundStyle(.blue)
                .frame(width: 32)
            VStack(alignment: .leading, spacing: 2) {
                Text(label).fontWeight(.medium)
                Text(subtitle).font(.caption).foregroundStyle(.secondary)
            }
            Spacer()
            Image(systemName: "chevron.right").foregroundStyle(.tertiary)
        }
        .padding(.vertical, 4)
    }
}
```

```swift
import SwiftUI
import BackupEngine

/// Setup view for a mounted volume destination (already mounted NAS or external drive).
struct MountedVolumeSetupView: View {
    @Environment(BackupCoordinator.self) private var coordinator
    @Environment(\.dismiss) private var dismiss

    @State private var name: String = ""
    @State private var selectedVolumeURL: URL? = nil
    @State private var volumes: [URL] = []
    @State private var testResult: TestResult = .idle
    @State private var isSaving = false

    enum TestResult {
        case idle
        case testing
        case success
        case failure(String)
    }

    var body: some View {
        Form {
            Section("Volume") {
                Picker("Volume", selection: $selectedVolumeURL) {
                    Text("Select a volume").tag(Optional<URL>.none)
                    ForEach(volumes, id: \.path) { url in
                        Text(volumeLabel(for: url)).tag(Optional(url))
                    }
                }
                .onChange(of: selectedVolumeURL) { _, newURL in
                    if let url = newURL, name.isEmpty {
                        name = url.lastPathComponent
                    }
                    testResult = .idle
                }
            }

            Section("Name") {
                TextField("Destination name", text: $name)
            }

            Section {
                Button("Test Connection") {
                    Task { await runTest() }
                }
                .disabled(selectedVolumeURL == nil || testResult == .testing)

                switch testResult {
                case .idle: EmptyView()
                case .testing: Label("Testing…", systemImage: "arrow.triangle.2.circlepath")
                case .success: Label("Connection successful", systemImage: "checkmark.circle.fill")
                        .foregroundStyle(.green)
                case .failure(let msg): Label(msg, systemImage: "xmark.circle.fill")
                        .foregroundStyle(.red)
                }
            }
        }
        .navigationTitle("Mounted Volume")
        .toolbar {
            ToolbarItem(placement: .confirmationAction) {
                Button("Save") { Task { await save() } }
                    .disabled(!canSave || isSaving)
            }
        }
        .task { loadVolumes() }
    }

    private var canSave: Bool {
        if case .success = testResult { return !name.isEmpty && selectedVolumeURL != nil }
        return false
    }

    private func loadVolumes() {
        volumes = FileManager.default.mountedVolumeURLs(
            includingResourceValuesForKeys: [.volumeNameKey, .volumeIsNetworkKey, .volumeIsRemovableKey],
            options: [.skipHiddenVolumes]
        ) ?? []
        // Filter out the boot volume ("/")
        volumes = volumes.filter { $0.path != "/" }
    }

    private func volumeLabel(for url: URL) -> String {
        let values = try? url.resourceValues(forKeys: [.volumeNameKey, .volumeIsNetworkKey])
        let volName = values?.volumeName ?? url.lastPathComponent
        let isNetwork = values?.volumeIsNetwork ?? false
        return isNetwork ? "\(volName) (Network)" : volName
    }

    private func runTest() async {
        guard let url = selectedVolumeURL else { return }
        testResult = .testing
        let adapter = MountedVolumeAdapter(config: DestinationConfig(
            id: UUID().uuidString,
            type: .mountedVolume,
            name: name.isEmpty ? "Test" : name,
            rootPath: url.path,
            createdAt: Date()
        ))
        let status = await adapter.probe()
        switch status {
        case .available: testResult = .success
        case .unavailable(let reason): testResult = .failure(reason)
        }
    }

    private func save() async {
        guard let url = selectedVolumeURL, canSave else { return }
        isSaving = true
        let config = DestinationConfig(
            id: UUID().uuidString,
            type: .mountedVolume,
            name: name,
            rootPath: url.path,
            createdAt: Date()
        )
        // Persist to GRDB via coordinator
        if let db = coordinator.database {
            try? await db.pool.write { database in try config.save(database) }
        }
        coordinator.destinationManager.addDestination(config: config)
        isSaving = false
        dismiss()
    }
}
```

```swift
import SwiftUI
import BackupEngine

/// Setup view for direct SMB/NFS network drive with credentials stored in Keychain.
struct SMBSetupView: View {
    @Environment(BackupCoordinator.self) private var coordinator
    @Environment(\.dismiss) private var dismiss

    @State private var name: String = ""
    @State private var host: String = ""
    @State private var share: String = ""
    @State private var username: String = ""
    @State private var password: String = ""
    @State private var testResult: TestResult = .idle
    @State private var isSaving = false

    enum TestResult: Equatable {
        case idle, testing, success
        case failure(String)
    }

    var body: some View {
        Form {
            Section("Connection") {
                TextField("Host", text: $host, prompt: Text("192.168.1.10 or nas.local"))
                    .onChange(of: host) { _, _ in testResult = .idle }
                TextField("Share name", text: $share, prompt: Text("Backups"))
                    .onChange(of: share) { _, _ in testResult = .idle }
            }

            Section("Credentials") {
                TextField("Username", text: $username)
                    .onChange(of: username) { _, _ in testResult = .idle }
                SecureField("Password", text: $password)
                    .onChange(of: password) { _, _ in testResult = .idle }
            }

            Section("Name") {
                TextField("Destination name", text: $name, prompt: Text("My NAS"))
            }

            Section {
                Button("Test Connection") {
                    Task { await runTest() }
                }
                .disabled(!canTest || testResult == .testing)

                switch testResult {
                case .idle: EmptyView()
                case .testing:
                    Label("Connecting… (up to 10 seconds)", systemImage: "arrow.triangle.2.circlepath")
                case .success:
                    Label("Connection successful", systemImage: "checkmark.circle.fill")
                        .foregroundStyle(.green)
                case .failure(let msg):
                    Label(msg, systemImage: "xmark.circle.fill")
                        .foregroundStyle(.red)
                        .lineLimit(3)
                }
            }
        }
        .navigationTitle("SMB / NFS Network Drive")
        .toolbar {
            ToolbarItem(placement: .confirmationAction) {
                Button("Save") { Task { await save() } }
                    .disabled(!canSave || isSaving)
            }
        }
    }

    private var canTest: Bool {
        !host.trimmingCharacters(in: .whitespaces).isEmpty &&
        !share.trimmingCharacters(in: .whitespaces).isEmpty &&
        !username.isEmpty && !password.isEmpty
    }

    private var canSave: Bool {
        if case .success = testResult { return canTest && !name.isEmpty }
        return false
    }

    private func runTest() async {
        testResult = .testing
        let mounter = NetFSMounter()
        do {
            let mountURL = try await mounter.mountShare(
                host: host.trimmingCharacters(in: .whitespaces),
                share: share.trimmingCharacters(in: .whitespaces),
                username: username,
                password: password
            )
            // Verify write access on the mount
            let adapter = MountedVolumeAdapter(config: DestinationConfig(
                id: UUID().uuidString, type: .nas, name: "test",
                rootPath: mountURL.path, createdAt: Date()
            ))
            let status = await adapter.probe()
            switch status {
            case .available: testResult = .success
            case .unavailable(let reason): testResult = .failure(reason)
            }
        } catch {
            testResult = .failure(error.localizedDescription)
        }
    }

    private func save() async {
        guard canSave else { return }
        isSaving = true

        let cleanHost = host.trimmingCharacters(in: .whitespaces)
        let cleanShare = share.trimmingCharacters(in: .whitespaces)

        // Mount to get rootPath (the /Volumes mount point)
        let mounter = NetFSMounter()
        do {
            let mountURL = try await mounter.mountShare(host: cleanHost, share: cleanShare, username: username, password: password)

            // Store credentials in Keychain — password NEVER goes to GRDB
            let store = KeychainCredentialStore()
            try store.save(server: cleanHost, account: username, password: password)

            let config = DestinationConfig(
                id: UUID().uuidString,
                type: .nas,
                name: name,
                rootPath: mountURL.path,
                createdAt: Date(),
                host: cleanHost,
                share: cleanShare,
                username: username
            )
            if let db = coordinator.database {
                try? await db.pool.write { database in try config.save(database) }
            }
            coordinator.destinationManager.addDestination(config: config)
            isSaving = false
            dismiss()
        } catch {
            testResult = .failure("Save failed: \(error.localizedDescription)")
            isSaving = false
        }
    }
}
```

```swift
import SwiftUI
import AppKit
import BackupEngine

/// Setup view for an iCloud Drive destination.
/// User picks any folder via NSOpenPanel — no iCloud API or entitlements required.
struct iCloudSetupView: View {
    @Environment(BackupCoordinator.self) private var coordinator
    @Environment(\.dismiss) private var dismiss

    @State private var name: String = ""
    @State private var selectedURL: URL? = nil
    @State private var testResult: TestResult = .idle
    @State private var isSaving = false

    enum TestResult: Equatable {
        case idle, testing, success
        case failure(String)
    }

    var body: some View {
        Form {
            Section("Folder") {
                HStack {
                    if let url = selectedURL {
                        Text(url.path)
                            .font(.caption)
                            .foregroundStyle(.secondary)
                            .lineLimit(2)
                            .truncationMode(.middle)
                    } else {
                        Text("No folder selected")
                            .foregroundStyle(.secondary)
                    }
                    Spacer()
                    Button("Choose…") {
                        if let url = pickFolder() {
                            selectedURL = url
                            if name.isEmpty { name = url.lastPathComponent }
                            testResult = .idle
                        }
                    }
                }
            }

            Section("Name") {
                TextField("Destination name", text: $name, prompt: Text("My iCloud Folder"))
            }

            Section {
                Button("Test Write Access") {
                    Task { await runTest() }
                }
                .disabled(selectedURL == nil || testResult == .testing)

                switch testResult {
                case .idle: EmptyView()
                case .testing: Label("Testing…", systemImage: "arrow.triangle.2.circlepath")
                case .success:
                    Label("Write access confirmed", systemImage: "checkmark.circle.fill")
                        .foregroundStyle(.green)
                case .failure(let msg):
                    Label(msg, systemImage: "xmark.circle.fill")
                        .foregroundStyle(.red)
                }
            }
            // Reminder about iCloud sync timing
            if case .success = testResult {
                Section {
                    Text("Note: iCloud upload may take time for large projects. Files are written locally first — sync happens in the background.")
                        .font(.caption)
                        .foregroundStyle(.secondary)
                }
            }
        }
        .navigationTitle("iCloud Drive")
        .toolbar {
            ToolbarItem(placement: .confirmationAction) {
                Button("Save") { Task { await save() } }
                    .disabled(!canSave || isSaving)
            }
        }
    }

    private var canSave: Bool {
        if case .success = testResult { return !name.isEmpty && selectedURL != nil }
        return false
    }

    private func pickFolder() -> URL? {
        let panel = NSOpenPanel()
        panel.canChooseFiles = false
        panel.canChooseDirectories = true
        panel.allowsMultipleSelection = false
        panel.message = "Choose a folder in iCloud Drive for backups"
        panel.prompt = "Choose"
        // Default to iCloud Drive root
        panel.directoryURL = URL(fileURLWithPath: NSHomeDirectory())
            .appendingPathComponent("Library/Mobile Documents/com~apple~CloudDocs")
        return panel.runModal() == .OK ? panel.url : nil
    }

    private func runTest() async {
        guard let url = selectedURL else { return }
        testResult = .testing
        let adapter = iCloudDestinationAdapter(config: DestinationConfig(
            id: UUID().uuidString, type: .icloud, name: name.isEmpty ? "test" : name,
            rootPath: url.path, createdAt: Date()
        ))
        let status = await adapter.probe()
        switch status {
        case .available: testResult = .success
        case .unavailable(let reason): testResult = .failure(reason)
        }
    }

    private func save() async {
        guard let url = selectedURL, canSave else { return }
        isSaving = true
        let config = DestinationConfig(
            id: UUID().uuidString,
            type: .icloud,
            name: name,
            rootPath: url.path,
            createdAt: Date()
        )
        if let db = coordinator.database {
            try? await db.pool.write { database in try config.save(database) }
        }
        coordinator.destinationManager.addDestination(config: config)
        isSaving = false
        dismiss()
    }
}
```
