**ALSParser.swift** — gzip decompress + XPath extract + classify:

```swift
import Foundation
import zlib

public enum ALSParserError: Error {
    case gzipInitFailed(Int32)
    case gzipDecompressFailed(Int32)
    case gzipCompressFailed(Int32)
}

public struct ALSParser {

    public enum ParseResult {
        case success(external: [URL], internal_: [URL])
        case parseFailure(reason: String)
    }

    public static func parse(alsURL: URL, projectDirectory: URL) -> ParseResult {
        // 1. Read raw bytes; check gzip magic 0x1F 0x8B
        guard let compressedData = try? Data(contentsOf: alsURL),
              compressedData.count >= 2,
              compressedData[0] == 0x1F, compressedData[1] == 0x8B else {
            return .parseFailure(reason: "Could not read .als file or not a gzip file")
        }
        // 2. Decompress using system zlib (NOT Apple NSData.decompressed — that is DEFLATE, not gzip)
        guard let xmlData = try? decompressGzip(compressedData) else {
            return .parseFailure(reason: "gzip decompression failed")
        }
        // 3. Parse XML with XMLDocument (DOM required for later path mutation)
        guard let doc = try? XMLDocument(data: xmlData, options: []) else {
            return .parseFailure(reason: "XML parse failed")
        }
        // 4. XPath: ALL SampleRef/FileRef/Path regardless of parent context
        guard let pathNodes = try? doc.nodes(forXPath: "//SampleRef/FileRef/Path") else {
            return .parseFailure(reason: "XPath query failed")
        }
        // 5. Extract non-empty Path Value attributes (Live 11+ plain-text format)
        // Skip empty Path Value — pre-Live 11 binary hex format; treat as unresolvable
        let allPaths: [URL] = pathNodes.compactMap { node in
            guard let el = node as? XMLElement,
                  let value = el.attribute(forName: "Value")?.stringValue,
                  !value.isEmpty else { return nil }
            return URL(fileURLWithPath: value).resolvingSymlinksInPath()
        }
        // 6. Classify: external = path NOT under project folder
        let resolvedProject = projectDirectory.resolvingSymlinksInPath().path
        var external: [URL] = []
        var internal_: [URL] = []
        for url in allPaths {
            if url.path.hasPrefix(resolvedProject + "/") || url.path == resolvedProject {
                internal_.append(url)
            } else {
                external.append(url)
            }
        }
        return .success(external: external, internal_: internal_)
    }

    // MARK: - Gzip decompress (zlib inflateInit2_ with MAX_WBITS + 32 = auto-detect gzip)
    public static func decompressGzip(_ data: Data) throws -> Data {
        var stream = z_stream()
        let initResult = inflateInit2_(&stream, MAX_WBITS + 32, ZLIB_VERSION, Int32(MemoryLayout<z_stream>.size))
        guard initResult == Z_OK else { throw ALSParserError.gzipInitFailed(initResult) }
        defer { inflateEnd(&stream) }

        var output = Data(count: max(data.count * 4, 65536))
        var status: Int32 = Z_OK

        data.withUnsafeBytes { inputPtr in
            stream.next_in = UnsafeMutablePointer(mutating: inputPtr.bindMemory(to: Bytef.self).baseAddress!)
            stream.avail_in = uInt(data.count)
            while status == Z_OK {
                output.withUnsafeMutableBytes { outputPtr in
                    stream.next_out = outputPtr.bindMemory(to: Bytef.self).baseAddress!
                        .advanced(by: Int(stream.total_out))
                    stream.avail_out = uInt(output.count) - uInt(stream.total_out)
                    status = inflate(&stream, Z_SYNC_FLUSH)
                }
                if stream.avail_out == 0 { output.count *= 2 }
            }
        }
        guard status == Z_STREAM_END else { throw ALSParserError.gzipDecompressFailed(status) }
        output.count = Int(stream.total_out)
        return output
    }
}
```

**ALSRewriter.swift** — in-memory XMLDocument path mutation + gzip re-compress + write:

```swift
import Foundation
import zlib

public struct ALSRewriter {

    /// Destination path for an external sample under Samples/Imported/.
    /// Full original absolute path preserved as subfolder structure (no collisions possible).
    /// Input:  /Users/eric/Music/Drums/kick.wav
    /// Output: Samples/Imported/Users/eric/Music/Drums/kick.wav
    public static func importedRelativePath(for sampleURL: URL) -> String {
        let path = sampleURL.path  // "/Users/eric/Music/Drums/kick.wav"
        let stripped = path.hasPrefix("/") ? String(path.dropFirst()) : path
        return "Samples/Imported/" + stripped
    }

    /// Rewrite all external sample FileRef paths in the XMLDocument to point to their
    /// Samples/Imported/<full-path> location in the backup, then gzip-compress and write
    /// the result to outputURL.
    ///
    /// - Parameters:
    ///   - alsURL: Original .als file (used to load and parse the XMLDocument)
    ///   - externalSamples: External sample URLs that were collected (exist on disk and were copied)
    ///   - backupProjectURL: Root of the backed-up project folder (contains Samples/Imported/)
    ///   - outputURL: Destination path where the rewritten .als should be written
    public static func rewriteAndCompress(
        alsURL: URL,
        externalSamples: [URL],
        backupProjectURL: URL,
        outputURL: URL
    ) throws {
        // 1. Load + decompress + parse (same as ALSParser but we need the mutable doc)
        guard let compressedData = try? Data(contentsOf: alsURL) else {
            throw ALSRewriterError.cannotReadSource
        }
        let xmlData = try ALSParser.decompressGzip(compressedData)
        let doc = try XMLDocument(data: xmlData, options: [])

        // 2. Build lookup: absolute path string -> new relative path from .als
        // The .als lives at <backupProjectURL>/<name>.als
        // The Samples/Imported/ folder is at <backupProjectURL>/Samples/Imported/
        // Relative path from .als to Samples/Imported/<...> = Samples/Imported/<...>
        // (since the .als is at the project root, not inside a subfolder)
        var pathMapping: [String: String] = [:]
        for url in externalSamples {
            let relPath = importedRelativePath(for: url)
            pathMapping[url.path] = relPath
        }

        // 3. Mutate FileRef elements
        let fileRefNodes = (try? doc.nodes(forXPath: "//SampleRef/FileRef")) ?? []
        for node in fileRefNodes {
            guard let fileRef = node as? XMLElement else { continue }
            guard let pathEl = fileRef.elements(forName: "Path").first,
                  let pathValue = pathEl.attribute(forName: "Value")?.stringValue,
                  !pathValue.isEmpty,
                  let relPath = pathMapping[pathValue] else { continue }

            // New absolute path in backup
            let newAbsPath = backupProjectURL.appendingPathComponent(relPath).path
            pathEl.attribute(forName: "Value")?.stringValue = newAbsPath

            // New relative path from .als location
            if let relPathEl = fileRef.elements(forName: "RelativePath").first {
                relPathEl.attribute(forName: "Value")?.stringValue = relPath
            }
            // RelativePathType = 1 (relative-to-document)
            if let relTypeEl = fileRef.elements(forName: "RelativePathType").first {
                relTypeEl.attribute(forName: "Value")?.stringValue = "1"
            }
        }

        // 4. Serialize with nodePreserveAll to keep original XML declaration and structure
        let rewrittenXML = doc.xmlData(options: .nodePreserveAll)

        // 5. Re-gzip with deflateInit2_ (MAX_WBITS + 16 = gzip format, not raw DEFLATE)
        let compressed = try compressGzip(rewrittenXML)

        // 6. Write to output location
        try compressed.write(to: outputURL)
    }

    // MARK: - Gzip compress (deflateInit2_ with MAX_WBITS + 16 = gzip header/trailer)
    static func compressGzip(_ data: Data) throws -> Data {
        var stream = z_stream()
        let initResult = deflateInit2_(&stream, 6, Z_DEFLATED, MAX_WBITS + 16, 8, Z_DEFAULT_STRATEGY,
                                       ZLIB_VERSION, Int32(MemoryLayout<z_stream>.size))
        guard initResult == Z_OK else { throw ALSParserError.gzipCompressFailed(initResult) }
        defer { deflateEnd(&stream) }

        var output = Data(count: data.count + 64)  // compressed <= original; +64 for gzip header
        data.withUnsafeBytes { inputPtr in
            stream.next_in = UnsafeMutablePointer(mutating: inputPtr.bindMemory(to: Bytef.self).baseAddress!)
            stream.avail_in = uInt(data.count)
            output.withUnsafeMutableBytes { outputPtr in
                stream.next_out = outputPtr.bindMemory(to: Bytef.self).baseAddress!
                stream.avail_out = uInt(output.count)
                _ = deflate(&stream, Z_FINISH)
            }
        }
        output.count = Int(stream.total_out)
        return output
    }
}

}
```

```swift
// Convenience: encode [URL] -> JSON string for storage
public static func encodePaths(_ urls: [URL]) -> String? {
    let paths = urls.map(\.path)
    return (try? JSONSerialization.data(withJSONObject: paths))
        .flatMap { String(data: $0, encoding: .utf8) }
}

// Convenience: decode JSON string -> [String] for display
public static func decodePaths(_ json: String?) -> [String] {
    guard let json, let data = json.data(using: .utf8),
          let array = try? JSONSerialization.jsonObject(with: data) as? [String] else {
        return []
    }
    return array
}
```
