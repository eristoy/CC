**DestinationStatusDot.swift** — reusable colored dot component:

```swift
import SwiftUI
import BackupEngine

/// A colored availability indicator dot with optional label.
/// Green = available, Red = unavailable.
struct DestinationStatusDot: View {
    let status: DestinationStatus

    var body: some View {
        Circle()
            .fill(dotColor)
            .frame(width: 8, height: 8)
            .help(helpText)
    }

    private var dotColor: Color {
        switch status {
        case .available:       return .green
        case .unavailable:     return .red
        }
    }

    private var helpText: String {
        switch status {
        case .available:               return "Available"
        case .unavailable(let reason): return "Unavailable: \(reason)"
        }
    }
}
```

import SwiftUI
import GRDB

struct DestinationsSettingsView: View {
    @Environment(BackupCoordinator.self) private var coordinator
    @Environment(DestinationManager.self) private var destinationManager
    @State private var destinations: [DestinationConfig] = []
    @State private var showAddSheet = false
    @State private var selectedDestID: String? = nil  // for error detail navigation

    var body: some View {
        VStack(spacing: 0) {
            if destinations.isEmpty {
                VStack(spacing: 8) {
                    Image(systemName: "externaldrive")
                        .font(.largeTitle)
                        .foregroundStyle(.secondary)
                    Text("No destinations configured")
                        .foregroundStyle(.secondary)
                    Button("Add Destination") { showAddSheet = true }
                        .buttonStyle(.bordered)
                }
                .frame(maxWidth: .infinity, maxHeight: .infinity)
                .padding()
            } else {
                List(selection: $selectedDestID) {
                    ForEach(destinations, id: \.id) { dest in
                        destinationRow(dest)
                            .tag(dest.id)
                            .swipeActions(edge: .trailing, allowsFullSwipe: false) {
                                Button(role: .destructive) {
                                    Task { await removeDestination(dest) }
                                } label: {
                                    Label("Remove", systemImage: "trash")
                                }
                            }
                    }
                }
                .listStyle(.inset)
            }

            Divider()
            HStack {
                Button(action: { showAddSheet = true }) {
                    Label("Add Destination", systemImage: "plus")
                }
                .buttonStyle(.plain)
                .padding(8)

                if let selID = selectedDestID,
                   let dest = destinations.first(where: { $0.id == selID }) {
                    Button(action: { Task { await removeDestination(dest) } }) {
                        Label("Remove", systemImage: "minus")
                    }
                    .buttonStyle(.plain)
                    .padding(.vertical, 8)
                }
                Spacer()
            }
            .background(.background)
        }
        .sheet(isPresented: $showAddSheet, onDismiss: { Task { await loadDestinations() } }) {
            AddDestinationView()
        }
        .task { await loadDestinations() }
        // Re-load when DestinationManager statuses change (probe results update)
        .onChange(of: destinationManager.statuses) { _, _ in
            // statuses updated — redraw rows with new dot colors (List already reactive to @Observable)
        }
        // Error detail sheet for tapped red-dot destination
        .sheet(item: Binding(
            get: { destinations.first(where: { $0.id == selectedDestID && isOffline($0) }) },
            set: { if $0 == nil { selectedDestID = nil } }
        )) { dest in
            DestinationErrorDetailView(destination: dest, destinationManager: destinationManager)
        }
    }

    // MARK: - Destination row

    @ViewBuilder
    private func destinationRow(_ dest: DestinationConfig) -> some View {
        let status = destinationManager.statuses[dest.id] ?? .unavailable(reason: "Unknown")
        HStack(spacing: 12) {
            Image(systemName: iconName(for: dest.type))
                .font(.title2)
                .foregroundStyle(.blue)
                .frame(width: 32)

            VStack(alignment: .leading, spacing: 2) {
                HStack(spacing: 6) {
                    Text(dest.name).fontWeight(.medium)
                    DestinationStatusDot(status: status)
                }
                Text(statusLabel(status: status, config: dest))
                    .font(.caption)
                    .foregroundStyle(.secondary)
                Text(dest.rootPath)
                    .font(.caption2)
                    .foregroundStyle(.tertiary)
                    .lineLimit(1)
                    .truncationMode(.middle)
            }
        }
        .padding(.vertical, 4)
        .contentShape(Rectangle())
        .onTapGesture {
            if isOffline(dest) { selectedDestID = dest.id }
        }
    }

    // MARK: - Helpers

    private func statusLabel(status: DestinationStatus, config: DestinationConfig) -> String {
        switch status {
        case .available:
            if let lastAt = config.lastConnectedAt {
                return "Online • Connected \(lastAt.formatted(.relative(presentation: .named)))"
            }
            return "Online"
        case .unavailable(let reason):
            return "Offline • \(reason)"
        }
    }

    private func isOffline(_ dest: DestinationConfig) -> Bool {
        if case .unavailable = destinationManager.statuses[dest.id] ?? .unavailable(reason: "") { return true }
        return false
    }

    private func iconName(for type: DestinationType) -> String {
        switch type {
        case .local:          return "externaldrive.fill"
        case .mountedVolume:  return "network"
        case .nas:            return "server.rack"
        case .icloud:         return "icloud.fill"
        case .github:         return "chevron.left.forwardslash.chevron.right"
        }
    }

    private func loadDestinations() async {
        guard let db = coordinator.database else { return }
        let results = try? await db.pool.read { database in
            try DestinationConfig.order(Column("createdAt").asc).fetchAll(database)
        }
        destinations = results ?? []
    }

    private func removeDestination(_ dest: DestinationConfig) async {
        guard let db = coordinator.database else { return }
        try? await db.pool.write { database in try DestinationConfig.deleteOne(database, key: dest.id) }
        destinationManager.removeDestination(id: dest.id)
        destinations.removeAll { $0.id == dest.id }
        if selectedDestID == dest.id { selectedDestID = nil }
    }
}

// MARK: - DestinationErrorDetailView

/// Shown when user taps a red-dot (offline) destination.
struct DestinationErrorDetailView: View {
    let destination: DestinationConfig
    let destinationManager: DestinationManager
    @Environment(\.dismiss) private var dismiss
    @State private var isRetrying = false
    @State private var retryResult: String? = nil

    var body: some View {
        VStack(alignment: .leading, spacing: 16) {
            HStack {
                Image(systemName: "xmark.circle.fill")
                    .foregroundStyle(.red)
                    .font(.title)
                VStack(alignment: .leading) {
                    Text(destination.name).font(.headline)
                    Text("Destination unavailable")
                        .foregroundStyle(.secondary)
                }
                Spacer()
                Button("Done") { dismiss() }
            }

            if let errorMsg = destination.lastErrorMessage {
                GroupBox("Last Error") {
                    Text(errorMsg)
                        .font(.caption)
                        .foregroundStyle(.secondary)
                        .frame(maxWidth: .infinity, alignment: .leading)
                }
            }

            if let lastConnected = destination.lastConnectedAt {
                Text("Last connected: \(lastConnected.formatted(.relative(presentation: .named)))")
                    .font(.caption)
                    .foregroundStyle(.secondary)
            }

            if let result = retryResult {
                Text(result)
                    .font(.caption)
                    .foregroundStyle(result.hasPrefix("Success") ? .green : .red)
            }

            Button(isRetrying ? "Retrying…" : "Retry Now") {
                Task { await retry() }
            }
            .buttonStyle(.bordered)
            .disabled(isRetrying)

            Spacer()
        }
        .padding()
        .frame(minWidth: 360, minHeight: 260)
    }

    private func retry() async {
        isRetrying = true
        retryResult = nil
        guard let adapter = destinationManager.adapter(for: destination.id) else {
            retryResult = "Adapter not found"
            isRetrying = false
            return
        }
        let status = await adapter.probe()
        switch status {
        case .available:
            destinationManager.statuses[destination.id] = .available
            retryResult = "Success — destination is online"
        case .unavailable(let reason):
            destinationManager.statuses[destination.id] = .unavailable(reason: reason)
            retryResult = "Still unavailable: \(reason)"
        }
        isRetrying = false
    }
}
