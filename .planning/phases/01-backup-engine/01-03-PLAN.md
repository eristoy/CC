<!-- Checksum strategy decision (from research) -->
Use CryptoKit SHA256 incremental hashing. xxHash-Swift may or may not be available
(plan 01-01 determined this). Check 01-01-SUMMARY.md. If xxHash is available, use it
(faster for large audio files). If not, use CryptoKit SHA256. Either way:
- Use the incremental (.update(data:)) API fed during the copy loop
- No second read pass after copy
- Store result as lowercase hex string

<!-- copyfile API -->
COPYFILE_CLONE (auto-fallback):
  #include <copyfile.h>
  copyfile(src, dst, NULL, COPYFILE_CLONE | COPYFILE_ALL)
  - Returns 0 on success (clone or fallback copy)
  - Returns non-zero on error
  - On APFS: near-instant clone, space-efficient
  - On ExFAT/HFS+: falls back to regular copy automatically
  DO NOT use COPYFILE_CLONE_FORCE — it fails on non-APFS without fallback

After clone succeeds: compute checksum from destination file (separate read — clone is instant so one read is acceptable here).
After chunked copy: checksum computed inline (no extra read needed).

import Foundation
import CryptoKit

// If xxHash-Swift was included (check 01-01-SUMMARY.md): import xxHash_Swift
// and use XXH64 instead of SHA256 for performance on large audio files.

enum FileCopyPipeline {
    /// Copies a file from source to destination.
    /// Attempts APFS clone (COPYFILE_CLONE) first; falls back to chunked copy.
    /// Returns the hex-encoded checksum of the destination file.
    static func copyFileWithChecksum(
        source: URL,
        destination: URL
    ) throws -> String {
        // Ensure destination directory exists
        let destDir = destination.deletingLastPathComponent()
        try FileManager.default.createDirectory(at: destDir, withIntermediateDirectories: true)

        // Attempt APFS clone (auto-fallback on non-APFS)
        let cloneResult = source.withUnsafeFileSystemRepresentation { srcPath in
            destination.withUnsafeFileSystemRepresentation { dstPath in
                copyfile(srcPath, dstPath, nil, copyfile_flags_t(COPYFILE_CLONE | COPYFILE_ALL))
            }
        }

        if cloneResult == 0 {
            // Clone succeeded — compute checksum from destination (one read, clone was instant)
            return try computeChecksum(of: destination)
        }

        // Clone failed (non-APFS or cross-device) — chunked copy with inline hashing
        return try chunkedCopyWithChecksum(source: source, destination: destination)
    }

    static func computeChecksum(of url: URL) throws -> String {
        var hasher = SHA256()
        let handle = try FileHandle(forReadingFrom: url)
        defer { try? handle.close() }
        let chunkSize = 8 * 1024 * 1024  // 8 MB
        while true {
            let chunk = try handle.read(upToCount: chunkSize)
            guard !chunk.isEmpty else { break }
            hasher.update(data: chunk)
        }
        return hasher.finalize().map { String(format: "%02hhx", $0) }.joined()
    }

    private static func chunkedCopyWithChecksum(source: URL, destination: URL) throws -> String {
        // ... chunked read/write with inline hashing
    }
}

import Foundation

public struct LocalDestinationAdapter: DestinationAdapter {
    public let id: String
    public let config: DestinationConfig

    public init(config: DestinationConfig) {
        self.id = config.id
        self.config = config
    }

    public func transfer(
        _ files: [FileEntry],
        versionID: String,
        progress: @escaping @Sendable (TransferProgress) -> Void
    ) async throws -> [BackupFileRecord] {
        let rootURL = URL(fileURLWithPath: config.rootPath)
        // Version directory: {rootPath}/{projectSlug}/{versionID}/
        // projectSlug derived from context — BackupEngine passes it via versionID prefix or separate param
        // For Phase 1: use the versionID directly as the version directory name under rootPath
        // Full path: {rootPath}/{versionID}/
        let versionDir = rootURL.appendingPathComponent(versionID)
        try FileManager.default.createDirectory(at: versionDir, withIntermediateDirectories: true)

        var records: [BackupFileRecord] = []
        for (index, file) in files.enumerated() {
            let destURL = versionDir.appendingPathComponent(file.relativePath)
            let checksum = try FileCopyPipeline.copyFileWithChecksum(
                source: file.url, destination: destURL)
            records.append(BackupFileRecord(
                rowid: nil,
                versionID: versionID,
                relativePath: file.relativePath,
                sourceMtime: file.mtime,
                sourceSize: file.size,
                checksum: checksum,
                copied: true
            ))
            progress(TransferProgress(
                relativePath: file.relativePath,
                bytesCopied: file.size,
                totalFiles: files.count,
                completedFiles: index + 1
            ))
        }
        return records
    }

    public func pruneVersions(beyond retentionCount: Int, for projectID: String) async throws {
        // Pruning is orchestrated by VersionManager (plan 01-04).
        // LocalDestinationAdapter exposes the ability to delete a version directory:
        // This method is a stub for Phase 1 — VersionManager calls it after marking status=deleting.
        // Implementation: list subdirectories of rootPath sorted by name (ISO8601-sortable),
        // delete directories that are marked for deletion by VersionManager.
        // TODO: implement when VersionManager (01-04) defines the pruning contract.
        // For now: no-op (safe — nothing is deleted until VersionManager confirms it).
    }

    public func probe() async -> DestinationStatus {
        let rootURL = URL(fileURLWithPath: config.rootPath)
        let exists = FileManager.default.fileExists(atPath: rootURL.path)
        return exists ? .available : .unavailable(reason: "Directory does not exist: \(config.rootPath)")
    }
}
